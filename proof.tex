	\begin{frame}{Konstrunkcja Macierzy Danych}
		Na początku zdefiniujemy, jak wyglądają nasze dane. Wektory $X_i$, o wymiarach $d\times 1$, reprezentują obserwacje o d cechach. Natomiast $\mathbb{X}$ ($n\times d$) jest tablicą składającą się z transponowanych wektorów $X_i$.
		\vspace{10mm}
		\begin{equation*}
			X_i = \begin{bmatrix}
				x_1\\
				x_2\\
				\vdots\\
				x_d
			\end{bmatrix}\quad\quad
			\mathbb{X} = \begin{bmatrix}
				\text{---} & X_1^T & \text{---}\\
				\text{---} &X_2^T& \text{---}\\ \
				&\vdots&\\
				\text{---} &X_n^T& \text{---}
			\end{bmatrix}
		\end{equation*}
		
	\end{frame}

	\begin{frame}{Konstrukcja Macierzy Kowariancji}
		\begin{equation*}
			\text{Wzór na kowariancję: } \cov(X, Y) = \frac{1}{n} \sum_{i=1}^{n}(x_i - \bar{x})(y_i - \bar{y})
		\end{equation*}
		W dalszej części zakładamy, że zmienne są wyśrodkowane względem swoich średnich ($\bar{x} = \bar{y} = 0$).
		Wtedy w naszej macierzy $\mathbb{X}$, kowariancję j-tej i k-tej zmiennej możemy zapisać w postaci
		\begin{equation*}
			\cov(j ,k) = \frac{1}{n} \sum_{i=0}^{n} x_{ij}x_{ik}
		\end{equation*}
		Gdzie elemnt $x_{ij}$ jest i-tą obserwacją j-tej zmiennej
	\end{frame}
	
	\begin{frame}{Konstrukcja Macierzy Kowariancji}
		Przyjrzyjmy się teraz macierzy
		\begin{equation*}
			M = \mathbb{X}^T \mathbb{X}
		\end{equation*}
		a konkretnie elementowi znajdującemu się w j-tym wierszu i k-tej kolumnie, który oznaczymy jako $M_{jk}$. Ze wzoru na mnożenie macierzy wiemy, że jest to iloczyn skalarny k-tej kolumny $\mathbb{X}$ oraz j-tego wiersza $\mathbb{X}^T$, czyli j-tej kolumny $\mathbb{X}$. Dzięki temu otrzymujemy
		\begin{equation*}
			M_{jk} = x_{1j}x_{1k} + x_{2j}x_{2k} + \cdots + x_{nj}x_{nk} = \sum_{i=0}^{n} x_{ij}x_{ik} = n\cov(j, k)
		\end{equation*}
		Łatwo wtedy zauważyć, że macierz kowariancji $S$, to
		\begin{equation*}
			S = \frac{1}{n} M = \frac{1}{n} \mathbb{X}^T \mathbb{X}
		\end{equation*}
		
	\end{frame}
	
	\begin{frame}{Cel i Założenia}
		\textbf{Założenia:}\\
		Zbiór danych jest: 
		\begin{enumerate}
			\item wyśrodkowany: $\overline{X_i}=0$
			\item znormalizowany: wartości mieszczą się w przedziale [0, 1]
		\end{enumerate}\vspace{20pt}
		
		\textbf{Cel: }znalezienie wektora $\|w\|=1$, który maksymalizuje wariancję.
	\end{frame}
	
	\begin{frame}{Obliczanie Wariancji}
		Niech $Z = [z_1, z_2, \ldots, z_n]^T$, będzie rzutem ortogonalnym danych na wektor $w$. Wtedy
		
		\begin{equation*}
			\var(Z) = \frac{1}{n}\sum_{i=1}^{n}(z_i)^2 = \frac{1}{n}\sum_{i=1}^{n}(X_i^T w)^2 = \frac{1}{n} (\mathbb{X}w)^T\mathbb{X}w 
		\end{equation*}
		\begin{equation*}
			= w^T\left( \frac{1}{n} \mathbb{X}^T \mathbb{X} \right) w = w^T S {}w
			\vspace{2.5mm}
		\end{equation*}
		
	\end{frame}

	
	\begin{frame}{Optymalizacja z Użyciem Mnożników Lagrange'a}
		Naszym celem jest zmaksymalizowanie wariancji, $\var(Z) = w^TSw$, z ograniczeniem $\|w\| = 1 \implies w^Tw = 1$.
		\begin{equation*}
			\pdv{(w^TSw)}{w} = \lambda \pdv{(w^Tw - 1)}{w}
		\end{equation*}
		\begin{equation*}
			2Sw = 2\lambda w \implies Sw=\lambda w
		\end{equation*}
		
		Czyli $w$, jest wektorem własnym macierzy $S$. Pondato
		\begin{equation*}
			w^TSw = w^T\lambda w = \lambda
		\end{equation*}
		czyli jest to dokładnie wektor odpowiadający największe wartości własnej.
	\end{frame}
